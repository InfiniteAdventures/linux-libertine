\newcommand{\printFDFont}[1]{%
\begingroup
  \let\SAVDeclareFontShape=\DeclareFontShape
  \let\DeclareFontShape=\MyDeclareFontShape
  \immediate\openout\myfile=#1.inc
  \InputIfFileExists{#1.fd}{}{}
  \immediate\closeout\myfile
\endgroup
}

  \typeout{\string\DeclareFontShape=#1/#2/#3/#4}%
  \IfSubStringInString{ssub}{#5}{}{%
    \Write{\string\myfdentry\string{#1\string}\string{#2\string}\string{#3\string}\string{#4\string}}}
}
