% fxl  -> normal

% fxlj -> oldstyle
% fxljo -> oldstyle + old ss
% fxlf -> fitted
\input fontinst.sty
\recordtransforms{driver.recs}

\def\installopt#1#2#3#4#5#6#7#8#9{%
  \message{^^J    #1-#3     ^^J}%
  \installfont{#1-#3}{#2-8r,#2-xl-8x,#2-xl-t1,#4}{#5}  {#6}{#7}{#8}{#9}{<-> \string\fxl@@scale}%
}
% -------------------------------------------------------------------

\installfonts

% regular
\installopt{fxlr}{fxlr}{t1jo}{libertine option ss}{t1os}           {T1}{fxljo}{m}{n}

% regular slanted
\installopt{fxlro}{fxlro}{t1jo}{libertine option ss}{t1os}         {T1}{fxljo}{m}{sl}

% regular italic
\installopt{fxlri}{fxlri}{t1jo}{libertine option ss}{t1os}         {T1}{fxljo}{m}{it}

% bold
\installopt{fxlb}{fxlb}{t1jo}{libertine option ss}{t1os}           {T1}{fxljo}{b}{n}

% bold slanted
\installopt{fxlbo}{fxlbo}{t1jo}{libertine option ss}{t1os}         {T1}{fxljo}{b}{sl}

% bold italic
\installopt{fxlbi}{fxlbi}{t1jo}{libertine option ss}{t1os}         {T1}{fxljo}{b}{it}

% small caps regular
\installopt{fxlrc}{fxlr}{t1jo}{libertine option sssc}{t1scos}      {T1}{fxljo}{m}{sc}

% % small caps bold
\installopt{fxlbc}{fxlb}{t1jo}{libertine option sssc}{t1scos}      {T1}{fxljo}{b}{sc}

% small caps italic
\installopt{fxlric}{fxlri}{t1jo}{libertine option sssc}{t1scos}    {T1}{fxljo}{m}{ic}

% small caps bold italic
\installopt{fxlbic}{fxlbi}{t1jo}{libertine option sssc}{t1scos}    {T1}{fxljo}{b}{ic}

% ---------------------------------------------------
\endinstallfonts
\endrecordtransforms
\input finstmsc.sty
\resetstr{PSfontsuffix}{.pfb}
\input xldriver.inc
\addDriver{t1jo}
\input driver.recs
\donedrivers
\bye
