\def\installopt#1#2#3#4#5#6#7#8#9{%
  \message{^^J    #1-#3     ^^J}%
  \installfont{#1-#3}{#2-8r,#2-xb-8x,#2-xb-e0,#2-xb-t1,#4}{#5}  {#6}{#7}{#8}{#9}{<-> \string\fxl@@scale}%
}

% bold
\installopt{fxbb}{fxbb}{t1o}{biolinum option ss}{t1}            {T1}{fxbo}{b}{n}

% bold slanted
