\input fontinst.sty
\recordtransforms{driver.recs}

\def\installopt#1#2#3#4#5#6{%
  \message{^^J    #1-#2     ^^J}%
  \installrawfont{#1-#2}{#1-#2}{#2}     {#3}{#4}{#5}{#6}{<-> \string\fxl@@scale}%
}
% -------------------------------------------------------------------

\foreach(uninumber){00,01,02,03,04,05,06,07,09,1e,1f,%
	20,21,22,23,24,25,26,27,2c,a7,e0,e1,f6,fb,ff}%
   \installfonts
     \installopt{fxlr}{xl-\str{uninumber}}  {U}{fxl\str{uninumber}}{m}{n}
     \installopt{fxlro}{xl-\str{uninumber}}  {U}{fxl\str{uninumber}}{m}{sl}
     \installopt{fxlri}{xl-\str{uninumber}} {U}{fxl\str{uninumber}}{m}{it}
     \installopt{fxlb}{xl-\str{uninumber}}  {U}{fxl\str{uninumber}}{b}{n}
     \installopt{fxlbo}{xl-\str{uninumber}}  {U}{fxl\str{uninumber}}{b}{sl}
     \installopt{fxlbi}{xl-\str{uninumber}} {U}{fxl\str{uninumber}}{b}{it}
   \endinstallfonts
\endfor(uninumber)

% ---------------------------------------------------
\endrecordtransforms
\input finstmsc.sty
\resetstr{PSfontsuffix}{.pfb}
\input xldriver.inc
\addDriver{unicode}
\input driver.recs
\donedrivers
\bye
