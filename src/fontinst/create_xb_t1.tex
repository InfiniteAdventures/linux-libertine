\input fontinst.sty
\recordtransforms{driver.recs}

\def\installopt#1#2#3#4#5#6#7#8#9{%
  \message{^^J    #1-#3     ^^J}%
  \installfont{#1-#3}{#2-8r,#2-xb-8x,#2-xb-t1,#4}{#5}  {#6}{#7}{#8}{#9}{<-> \string\fxl@@scale}%
}
% -------------------------------------------------------------------

\installfonts

% regular
\installopt{fxbr}{fxbr}{t1}{biolinum}{t1}                  {T1}{fxb}{m}{n}

% italic
\installopt{fxbri}{fxbri}{t1}{biolinum}{t1}                {T1}{fxb}{m}{it}

% regular slanted
\installopt{fxbro}{fxbro}{t1}{biolinum}{t1}                {T1}{fxb}{m}{sl}

% bold
\installopt{fxbb}{fxbb}{t1}{biolinum}{t1}                  {T1}{fxb}{b}{n}

% bold slanted
\installopt{fxbbo}{fxbbo}{t1}{biolinum}{t1}                {T1}{fxb}{b}{sl}

% small caps regular
\installopt{fxbrc}{fxbr}{t1}{biolinum option sc}{t1sc}     {T1}{fxb}{m}{sc}

% small caps bold
\installopt{fxbbc}{fxbb}{t1}{biolinum option sc}{t1sc}     {T1}{fxb}{b}{sc}

% ---------------------------------------------------
\endinstallfonts
\endrecordtransforms
